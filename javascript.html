تعلم JavaScript بالدارجة المغربية

شنو هي JavaScript؟

JavaScript هي لغة برمجة كتخلي صفحات الويب ديالك ديناميكية وتفاعلية.
يعني تقدر تزيد حركات، تستقبل مدخلات المستخدم، تحسب وتحلل البيانات، وأكثر من هاد الشي.
بلا JavaScript، صفحات الويب كتكون ثابتة فقط (نصوص وصور). 

// هذا مثال بسيط بلغة JavaScript
console.log('السلام عليكم!');

كيفاش نكتب كود JavaScript؟

كود JavaScript كيكتب داخل الوسم <script> في الصفحة ديالك.
تقدر تكتبها مباشرة داخل ملف HTML أو في ملف خارجي .js. 

<script>
  alert('مرحبا بك في صفحة الويب!');
</script>

المتغيرات Variables

المتغير هو مكان كتخزن فيه بيانات مؤقتة.
في JavaScript تقدر تستعمل let أو const لتعريف المتغيرات. 

let age = 25;
const name = 'سعيد';

الدوال Functions

الدالة هي مجموعة من الأوامر اللي كتقوم بعمل معين، تقدر تعيد استعمالها مرات كثيرة. 

function sayHello() {
  alert('سلام!');
}
sayHello();

الشروط if else

كتستعمل الشروط باش تحدد واش كيدوز كود معين أو لا بناءً على حالة معينة. 

let age = 18;
if (age >= 18) {
  alert('مسموح لك تدخل');
} else {
  alert('ممنوع عليك');
}

الحلقات Loops

الحلقات كتسمحلك تكرر نفس الكود بزاف ديال المرات بشكل أوتوماتيكي. 

for (let i = 0; i < 5; i++) {
  console.log('العدد: ' + i);
}

المصفوفات Arrays

المصفوفة هي مجموعة من القيم كتخزنها فمتغير واحد. 

let fruits = ['تفاح', 'موز', 'برتقال'];
console.log(fruits[1]); // يطبع موز

الكائنات Objects

الكائن هو متغير كيخزن بيانات مركبة على شكل خصائص وقيم. 

let person = {
  name: 'ليلى',
  age: 30,
  city: 'الدار البيضاء'
};
console.log(person.name);

DOM - كيفاش نبدلو الصفحة بالـ JavaScript؟

DOM هي الوثيقة اللي كتحتوي على كل عناصر الصفحة، وJavaScript كتقدر تبدل فيها وتزيد تحكم فالصفحة.
مثلاً، تقدر تبدل النص، تضيف أزرار، ولا تحذف عناصر. 

<button id="btn">ضغظني</button>

const btn = document.getElementById('btn');
btn.addEventListener('click', function() {
  alert('تهلا ف راسك!');
});

تغيير المحتوى النصي

تقدر تبدل النص داخل أي عنصر بسهولة. 

const title = document.querySelector('h1');
title.textContent = 'مرحبا ب JavaScript!';

التعامل مع الأحداث Events

الأحداث هي ردود الفعل على أفعال المستخدم، بحال الضغط على زر، تحريك الماوس، أو كتابة نص.
JavaScript كتسمح لك تتعامل مع هاد الأحداث وتبرمج الردود ديالك. 

<input id="inputBox" type="text" placeholder="كتب شي حاجة">

const inputBox = document.getElementById('inputBox');
inputBox.addEventListener('input', function(event) {
  console.log(event.target.value);
});

تطبيق بسيط: حساب مجموع رقمين

دابا غادي نديروا مثال بسيط بحساب مجموع رقمين يدخلهم المستخدم. 


<input id="num1" type="number" placeholder="الرقم 1">
<input id="num2" type="number" placeholder="الرقم 2">
<button id="calc">حسب المجموع</button>
<div id="result"></div>

const num1 = document.getElementById('num1');
const num2 = document.getElementById('num2');
const calc = document.getElementById('calc');
const result = document.getElementById('result');

calc.addEventListener('click', function() {
  const sum = Number(num1.value) + Number(num2.value);
  result.textContent = 'المجموع هو: ' + sum;
});

نصائح مهمة لتعلم JavaScript

  • جرب تكتب الكود بيدك على المتصفح أو محرر الكود.
  • حاول تفهم كل سطر كتبته، وشنو كيدير.
  • شوف الدوكومنتيشن الرسمية بحال MDN JavaScript.
  • تعلم تنسيق الصفحة بـ CSS ودمجها مع JavaScript لتزيد في التحكم.

خلاصة طويلة وبسيطة

JavaScript هي اللغة اللي كتزيد الحياة والتفاعل لصفحات الويب.
من المتغيرات والدوال للحلقات والكائنات، تعلمها خطوة بخطوة كيخليك تبني مواقع قوية.
DOM هو المفتاح باش تتحكم في الصفحة بعد تحميلها.
التمرين المستمر والتجريب هو سر النجاح.
تعلم JavaScript كيبدا بفهم الأساسيات، ومن بعد كتزيد تعمق بالمشاريع.
